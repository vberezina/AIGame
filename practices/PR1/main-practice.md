```python
import pygame
import math

# –ù–∞—Å—Ç—Ä–æ–π–∫–∏ —ç–∫—Ä–∞–Ω–∞
WIDTH, HEIGHT = 800, 600
FPS = 60

class Agent:
    def __init__(self, x, y):
        self.pos = pygame.Vector2(x, y)
        self.speed = 3
        self.radius = 15

    def update(self, target_pos):
        # 1. –í—ã—á–∏—Å–ª—è–µ–º –≤–µ–∫—Ç–æ—Ä –∫ —Ü–µ–ª–∏
        target = pygame.Vector2(target_pos)
        direction = target - self.pos
        
        # 2. –î–≤–∏–≥–∞–µ–º—Å—è, —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –º—ã –Ω–µ –≤ —Å–∞–º–æ–π —Ü–µ–ª–∏
        if direction.length() > 0:
            direction = direction.normalize() # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è
            self.pos += direction * self.speed

    def draw(self, surface):
        pygame.draw.circle(surface, (0, 255, 0), (int(self.pos.x), int(self.pos.y)), self.radius)

def main():
    pygame.init()
    screen = pygame.display.set_mode((WIDTH, HEIGHT))
    clock = pygame.time.Clock()
    
    agent = Agent(WIDTH // 2, HEIGHT // 2)
    
    running = True
    while running:
        screen.fill((30, 30, 30)) # –¢–µ–º–Ω—ã–π —Ñ–æ–Ω
        
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False

        # –¶–µ–ª—å ‚Äî –∫—É—Ä—Å–æ—Ä –º—ã—à–∏
        mouse_pos = pygame.mouse.get_pos()
        
        # –û—Ç—Ä–∏—Å–æ–≤–∫–∞ —Ü–µ–ª–∏
        pygame.draw.circle(screen, (255, 0, 0), mouse_pos, 5)
        
        # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –∏ –æ—Ç—Ä–∏—Å–æ–≤–∫–∞ –∞–≥–µ–Ω—Ç–∞
        agent.update(mouse_pos)
        agent.draw(screen)

        pygame.display.flip()
        clock.tick(FPS)

    pygame.quit()

if __name__ == "__main__":
    main()

```
